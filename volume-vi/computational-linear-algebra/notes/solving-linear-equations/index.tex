\section{Solving Linear Equations}

\begin{toolkitbox}{Matrix Operations --- Quick Reference}
\begin{tabular}{@{}p{0.30\textwidth}p{0.63\textwidth}@{}}
\textbf{Associativity}    & $A(BC) = (AB)C$. \\[4pt]
\textbf{Distributivity}   & $A(B+C) = AB + AC$; $(A+B)C = AC + BC$. \\[4pt]
\textbf{NOT commutative}  & $AB \neq BA$ in general. \\[4pt]
\textbf{Identity}         & $AI = IA = A$. \\[4pt]
\textbf{Transpose product}& $(AB)^T = B^T A^T$. \\[4pt]
\textbf{Diagonal matrices}& $D_1 D_2$ is diagonal; $DA$ scales rows; $AD$ scales columns. \\
\end{tabular}
\end{toolkitbox}


\begin{toolkitbox}{Inverse Matrices --- Quick Reference}
\begin{tabular}{@{}p{0.30\textwidth}p{0.63\textwidth}@{}}
\textbf{Inverse}          & $A^{-1}$: the unique matrix with $AA^{-1} = A^{-1}A = I$. \\[4pt]
\textbf{Existence}        & $A^{-1}$ exists iff $A$ is square and $\det(A) \neq 0$ (nonsingular). \\[4pt]
\textbf{$2\times 2$ formula} & $\begin{pmatrix} a & b \\ c & d \end{pmatrix}^{-1} = \dfrac{1}{ad-bc}\begin{pmatrix} d & -b \\ -c & a \end{pmatrix}$. \\[8pt]
\textbf{Product rule}     & $(AB)^{-1} = B^{-1}A^{-1}$ (order reverses). \\[4pt]
\textbf{Transpose rule}   & $(A^T)^{-1} = (A^{-1})^T$. \\[4pt]
\textbf{Change of basis}  & If $M$ converts basis $B_1$ to $B_2$, then $M^{-1}$ converts $B_2$ to $B_1$. \\
\end{tabular}
\end{toolkitbox}
