\section{Vectors}

\begin{toolkitbox}{Vectors and Linear Combinations --- Quick Reference}
\begin{tabular}{@{}p{0.28\textwidth}p{0.65\textwidth}@{}}
\textbf{Vector in $\mathbb{R}^n$}     & An ordered list $(v_1, v_2, \ldots, v_n)$ of $n$ real numbers. \\[4pt]
\textbf{Addition}                      & $(v_1,\ldots,v_n)+(w_1,\ldots,w_n) = (v_1+w_1,\ldots,v_n+w_n)$. \\[4pt]
\textbf{Scalar multiplication}         & $c(v_1,\ldots,v_n) = (cv_1,\ldots,cv_n)$. \\[4pt]
\textbf{Linear combination}            & $c_1\mathbf{v}_1 + c_2\mathbf{v}_2 + \cdots + c_k\mathbf{v}_k$; scalars times vectors, summed. \\[4pt]
\textbf{Convex combination}            & Linear combination with $c_i \geq 0$ and $\sum c_i = 1$; result lies inside the convex hull. \\[4pt]
\textbf{Standard basis vectors}        & $\mathbf{e}_j \in \mathbb{R}^n$: all zeros except a $1$ in position $j$. Every vector is a linear combination of $\mathbf{e}_1,\ldots,\mathbf{e}_n$. \\
\end{tabular}
\end{toolkitbox}


\begin{toolkitbox}{Matrices --- Quick Reference}
\begin{tabular}{@{}p{0.28\textwidth}p{0.65\textwidth}@{}}
\textbf{Matrix}            & Rectangular array of numbers; $A \in \mathbb{R}^{m \times n}$ has $m$ rows, $n$ columns. \\[4pt]
\textbf{Matrix--vector}    & $(A\mathbf{x})_i = \sum_j A_{ij} x_j$; $A\mathbf{x}$ is a linear combination of columns of $A$. \\[4pt]
\textbf{Matrix--matrix}    & $(AB)_{ij} = \sum_k A_{ik} B_{kj}$; requires inner dimensions to match. \\[4pt]
\textbf{Transpose}         & $(A^T)_{ij} = A_{ji}$; rows become columns. \\[4pt]
\textbf{Identity}          & $I_n$: ones on diagonal, zeros elsewhere; $IA = AI = A$. \\[4pt]
\textbf{Column view}       & $A\mathbf{x} = x_1\mathbf{a}_1 + x_2\mathbf{a}_2 + \cdots + x_n\mathbf{a}_n$ where $\mathbf{a}_j$ are the columns of $A$. \\
\end{tabular}
\end{toolkitbox}


\begin{toolkitbox}{Lengths and Dot Products --- Quick Reference}
\begin{tabular}{@{}p{0.28\textwidth}p{0.65\textwidth}@{}}
\textbf{Dot product}     & $\mathbf{v} \cdot \mathbf{w} = v_1w_1 + v_2w_2 + \cdots + v_nw_n \in \mathbb{R}$. \\[4pt]
\textbf{Length (norm)}   & $\|\mathbf{v}\| = \sqrt{\mathbf{v} \cdot \mathbf{v}} = \sqrt{v_1^2 + \cdots + v_n^2}$. \\[4pt]
\textbf{Unit vector}     & $\hat{\mathbf{v}} = \mathbf{v}/\|\mathbf{v}\|$; has length 1. \\[4pt]
\textbf{Angle}           & $\cos\theta = \dfrac{\mathbf{v} \cdot \mathbf{w}}{\|\mathbf{v}\|\,\|\mathbf{w}\|}$. \\[6pt]
\textbf{Orthogonality}   & $\mathbf{v} \perp \mathbf{w}$ iff $\mathbf{v} \cdot \mathbf{w} = 0$. \\[4pt]
\textbf{Cauchy--Schwarz} & $|\mathbf{v} \cdot \mathbf{w}| \leq \|\mathbf{v}\|\,\|\mathbf{w}\|$. \\
\end{tabular}
\end{toolkitbox}
