\section{Vector Spaces}

\begin{toolkitbox}{Basis and Dimension --- Quick Reference}
\begin{tabular}{@{}p{0.30\textwidth}p{0.63\textwidth}@{}}
\textbf{Linearly independent} & No vector is a linear combination of the others; only $\mathbf{0} = c_1\mathbf{v}_1 + \cdots + c_k\mathbf{v}_k$ has $c_i = 0$ for all $i$. \\[4pt]
\textbf{Span}                 & $\operatorname{span}\{\mathbf{v}_1,\ldots,\mathbf{v}_k\}$: all linear combinations of $\mathbf{v}_1,\ldots,\mathbf{v}_k$. \\[4pt]
\textbf{Basis}                & A linearly independent set that spans the space. \\[4pt]
\textbf{Dimension}            & Number of vectors in any basis; all bases have the same size. \\[4pt]
\textbf{Change of basis}      & Transition matrix $P$: columns are the new basis vectors in terms of the old basis. \\
\end{tabular}
\end{toolkitbox}
