\section{Linear Transformations}

\begin{toolkitbox}{Standard Matrix --- Quick Reference}
\begin{tabular}{@{}p{0.30\textwidth}p{0.63\textwidth}@{}}
\textbf{Standard matrix}      & $[T] = [T(\mathbf{e}_1)\mid T(\mathbf{e}_2)\mid\cdots\mid T(\mathbf{e}_n)] \in \mathbb{R}^{m\times n}$. \\[4pt]
\textbf{How to build it}       & Evaluate $T$ on each standard basis vector; place results as columns. \\[4pt]
\textbf{Uniqueness}            & $[T]$ is the \emph{unique} matrix with $T(\mathbf{v}) = [T]\mathbf{v}$ for all $\mathbf{v}$. \\[4pt]
\textbf{Composition}           & $[S \circ T] = [S][T]$. \\[4pt]
\textbf{Change of basis}       & Same $T$, different bases $\Rightarrow$ different matrix; related by $B = P^{-1}AP$. \\
\end{tabular}
\end{toolkitbox}


\begin{toolkitbox}{Linear Transformations --- Quick Reference}
\begin{tabular}{@{}p{0.30\textwidth}p{0.63\textwidth}@{}}
\textbf{Definition}        & $T(c\mathbf{v} + d\mathbf{w}) = cT(\mathbf{v}) + dT(\mathbf{w})$ for all $\mathbf{v},\mathbf{w}$ and scalars $c,d$. \\[4pt]
\textbf{Origin fixed}      & Always $T(\mathbf{0}) = \mathbf{0}$; translation is \emph{not} linear. \\[4pt]
\textbf{Standard matrix}   & $[T] = [T(\mathbf{e}_1)\mid\cdots\mid T(\mathbf{e}_n)]$; then $T(\mathbf{v}) = [T]\mathbf{v}$. \\[4pt]
\textbf{Composition}       & $S \circ T$ has matrix $[S][T]$. \\[4pt]
\textbf{Affine map}        & $\Phi(\mathbf{r}) = A\mathbf{r} + \mathbf{v}$: linear part $A$ plus translation $\mathbf{v}$. \\[4pt]
\textbf{Invariance}        & B-spline curves: apply affine map to control points only. \\
\end{tabular}
\end{toolkitbox}
