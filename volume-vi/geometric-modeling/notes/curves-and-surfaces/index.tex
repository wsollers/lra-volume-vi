\section{Curves and Surfaces}

\begin{toolkitbox}{Parametric Curves and Surfaces --- Quick Reference}
\begin{tabular}{@{}p{0.28\textwidth}p{0.65\textwidth}@{}}
\textbf{Curve}               & $C(u):\,[a,b]\to\mathbb{R}^n$; vector-valued, one parameter. \\[4pt]
\textbf{Velocity}            & $C'(u)$; tangent direction and speed of traversal. \\[4pt]
\textbf{Unit tangent}        & $\hat{T}(u) = C'(u)/\|C'(u)\|$ at a regular point. \\[4pt]
\textbf{Surface}             & $S(u,v):\,R\to\mathbb{R}^3$; vector-valued, two parameters. \\[4pt]
\textbf{Partial derivatives} & $S_u = \partial S/\partial u$,\; $S_v = \partial S/\partial v$; tangent to isoparametric lines. \\[4pt]
\textbf{Unit normal}         & $N = (S_u\times S_v)/\|S_u\times S_v\|$ at a regular point. \\[4pt]
\textbf{Tangent plane}       & Spanned by $S_u$ and $S_v$ at the base point. \\[4pt]
\textbf{Directional deriv.}  & $D_{\mathbf{v}}S = \nabla S \cdot \hat{\mathbf{v}}$; rate of change in direction $\mathbf{v}$. \\
\end{tabular}
\end{toolkitbox}
