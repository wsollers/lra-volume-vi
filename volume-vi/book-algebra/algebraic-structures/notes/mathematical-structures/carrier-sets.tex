\subsection*{Carrier Sets}
\label{subsec:carrier-sets}

\begin{definition}[Carrier Set]
\label{def:carrier-set}
The \textbf{carrier set} of a structure is the underlying set on which the
structure's operations, relations, and distinguished elements are interpreted.
\end{definition}

\begin{remark*}[Standard quantified statement]
\[
\operatorname{CarrierSet}(A,\mathcal{A})
\Longleftrightarrow
A \text{ is the underlying set of } \mathcal{A}.
\]
\end{remark*}

\begin{remark*}[Definition predicate reading]
$A$ is the carrier set of $\mathcal{A}$ exactly when every symbol of
$\mathcal{A}$ is interpreted over $A$.
\end{remark*}

% Omitted Negated quantified statement: carrier-set failure is not meaningful

\begin{remark*}[Interpretation]
In a group $(G,\star)$, the carrier set is $G$.  In a ring $(R,+,\cdot)$, the
carrier set is $R$.  The additional symbols tell us what structure has been
placed on that set.
\end{remark*}
