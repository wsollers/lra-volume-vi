\begin{proposition*}[Multiplication by Additive Inverse]
Let $R$ be a ring. For all $a, b \in R$,
\[
a \cdot (-b) = -(a \cdot b)
\quad\text{and}\quad
(-a) \cdot b = -(a \cdot b).
\]
In any ring with unity, $(-1) \cdot a = -a$.
\end{proposition*}
