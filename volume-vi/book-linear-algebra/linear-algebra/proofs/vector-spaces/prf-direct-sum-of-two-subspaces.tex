\begin{theorem*}[Direct Sum of Two Subspaces]
Let $U$ and $W$ be subspaces of a vector space $V$.
Then
\[
U+W=U\oplus W
\]
if and only if
\[
U \cap W = \{0\}.
\]
\end{theorem*}
