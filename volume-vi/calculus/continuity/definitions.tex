\subsection*{Definitions}

\begin{definition*}[Continuity at a Point]
A function \(f\) is continuous at \(a\) when \(f(a)\) is defined,
\(\lim_{x\to a}f(x)\) exists, and
\[
  \lim_{x\to a}f(x)=f(a).
\]
\end{definition*}

\begin{definition*}[Continuity on an Interval]
A function is continuous on an interval when it is continuous at each interior
point and has the appropriate one-sided continuity at any endpoint included in
the interval.
\end{definition*}

\begin{definition*}[Common Discontinuities]
A removable discontinuity is a hole, a jump discontinuity has unequal
one-sided limiting values, and an infinite discontinuity occurs when function
values become unbounded near the point.
\end{definition*}
