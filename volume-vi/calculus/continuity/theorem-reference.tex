\subsection*{Theorem Reference}

\begin{theorem*}[Algebra of Continuous Functions]
Sums, differences, scalar multiples, products, quotients with nonzero
denominator, and compositions of continuous functions are continuous wherever
the resulting expressions are defined.
\end{theorem*}

% TODO: link to Volume II Intermediate Value Theorem.
\begin{theorem*}[Intermediate Value Theorem]
If \(f\) is continuous on \([a,b]\) and \(N\) lies between \(f(a)\) and
\(f(b)\), then there is some \(c\in[a,b]\) such that \(f(c)=N\).
\end{theorem*}
