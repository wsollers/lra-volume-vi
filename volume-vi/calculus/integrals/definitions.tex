\subsection*{Definitions}

\begin{definition*}[Antiderivative]
An antiderivative of \(f\) on an interval is a function \(F\) such that
\(F'=f\) on that interval.
\end{definition*}

\begin{definition*}[Definite Integral]
The definite integral \(\int_a^b f(x)\,dx\) measures signed accumulation of
\(f\) over \([a,b]\). Computationally, it is evaluated by antiderivatives
whenever the Fundamental Theorem of Calculus applies.
\end{definition*}

% TODO: link to Volume II Fundamental Theorem of Calculus.
\begin{theorem*}[Fundamental Theorem of Calculus]
If \(F'=f\) on an interval containing \(a\) and \(b\), then
\[
  \int_a^b f(x)\,dx = F(b)-F(a).
\]
\end{theorem*}
