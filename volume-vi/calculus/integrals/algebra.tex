\subsection*{Integral Algebra}

\begin{center}
\begin{tabular}{ll}
\toprule
Rule & Formula \\
\midrule
Linearity & \(\int (af+bg)=a\int f+b\int g\) \\
Interval additivity & \(\int_a^b f+\int_b^c f=\int_a^c f\) \\
Reversal & \(\int_a^b f=-\int_b^a f\) \\
Zero width & \(\int_a^a f=0\) \\
Constant multiple & \(\int_a^b c\,dx=c(b-a)\) \\
\bottomrule
\end{tabular}
\end{center}
