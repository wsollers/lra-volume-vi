\subsection*{Vector Calculus Theorems}

\begin{definition*}[Line and Surface Integrals]
Line integrals accumulate along curves. Surface integrals accumulate over
surfaces, often measuring flux through the surface.
\end{definition*}

% TODO: link to rigorous Green's theorem statement in the appropriate volume.
\begin{theorem*}[Green's Theorem]
For suitable planar regions and vector fields, a circulation integral around
the boundary equals a double integral of curl over the region.
\end{theorem*}

% TODO: link to rigorous Stokes' theorem statement in the appropriate volume.
\begin{theorem*}[Stokes' Theorem]
For suitable oriented surfaces, a circulation integral around the boundary
equals a surface integral of curl over the surface.
\end{theorem*}

% TODO: link to rigorous Divergence theorem statement in the appropriate volume.
\begin{theorem*}[Divergence Theorem]
For suitable solid regions, the flux across the boundary equals the triple
integral of divergence over the region.
\end{theorem*}
