\subsection*{Multiple Integrals}

\begin{definition*}[Double and Triple Integrals]
Double and triple integrals accumulate a quantity over a region in the plane
or in space:
\[
  \iint_R f\,dA,
  \qquad
  \iiint_E f\,dV.
\]
\end{definition*}

\begin{center}
\begin{tabular}{ll}
\toprule
Coordinate system & Common use \\
\midrule
Polar & Circular regions in the plane \\
Cylindrical & Circular symmetry around an axis \\
Spherical & Radial symmetry around a point \\
\bottomrule
\end{tabular}
\end{center}
