\begin{theorem*}[Limit Laws]
When the relevant limits exist, sums, differences, scalar multiples,
products, and quotients of functions have limits computed by applying the
same algebraic operation to the limiting values. Quotients require a
nonzero denominator limit.
\end{theorem*}

\begin{theorem*}[Squeeze Theorem]
If \(g(x)\le f(x)\le h(x)\) near \(a\) and
\[
  \lim_{x\to a}g(x)=\lim_{x\to a}h(x)=L,
\]
then \(\lim_{x\to a}f(x)=L\).
\end{theorem*}
