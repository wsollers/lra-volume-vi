\subsection*{Series}

\begin{definition*}[Series and Partial Sums]
The series \(\sum_{n=1}^{\infty} a_n\) is studied through its partial sums
\[
  s_N=\sum_{n=1}^{N}a_n.
\]
The series converges when the sequence \((s_N)\) converges.
\end{definition*}

\begin{center}
\begin{tabular}{ll}
\toprule
Series & Convergence pattern \\
\midrule
Geometric \(\sum ar^n\) & Converges for \(|r|<1\) \\
\(p\)-series \(\sum 1/n^p\) & Converges for \(p>1\) \\
\bottomrule
\end{tabular}
\end{center}
