\subsection*{Chain-Rule Patterns}

\begin{center}
\begin{tabular}{ll}
\toprule
Outer form & Pattern \\
\midrule
\((u(x))^n\) & \(n(u(x))^{n-1}u'(x)\) \\
\(e^{u(x)}\) & \(e^{u(x)}u'(x)\) \\
\(\ln(u(x))\) & \(u'(x)/u(x)\) \\
\(\sin(u(x))\) & \(\cos(u(x))u'(x)\) \\
\(\sqrt{u(x)}\) & \(u'(x)/(2\sqrt{u(x)})\) \\
\bottomrule
\end{tabular}
\end{center}

\begin{remark*}[Method]
When differentiating a composition, name the inside expression \(u\), apply
the derivative rule to the outside function, and multiply by \(u'\).
\end{remark*}
