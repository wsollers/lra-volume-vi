\subsection*{Derivative Algebra}

% TODO: link to Volume II linearity of derivative theorem.
\begin{theorem*}[Constant, Sum, Difference, and Scalar Multiple Rules]
For differentiable functions \(f\) and \(g\) and a constant \(c\),
\[
  (c)'=0,\qquad
  (f+g)'=f'+g',\qquad
  (f-g)'=f'-g',\qquad
  (cf)'=cf'.
\]
\end{theorem*}

% TODO: link to Volume II Product Rule theorem.
\begin{theorem*}[Product Rule]
If \(f\) and \(g\) are differentiable at \(x\), then
\[
  (fg)'(x)=f'(x)g(x)+f(x)g'(x).
\]
\end{theorem*}

% TODO: link to Volume II Quotient Rule theorem.
\begin{theorem*}[Quotient Rule]
If \(f\) and \(g\) are differentiable at \(x\) and \(g(x)\ne 0\), then
\[
  \left(\frac{f}{g}\right)'(x)
    =\frac{g(x)f'(x)-f(x)g'(x)}{g(x)^2}.
\]
\end{theorem*}

% TODO: link to Volume II Chain Rule theorem.
\begin{theorem*}[Chain Rule]
If \(g\) is differentiable at \(x\) and \(f\) is differentiable at \(g(x)\),
then
\[
  (f\circ g)'(x)=f'(g(x))g'(x).
\]
\end{theorem*}
