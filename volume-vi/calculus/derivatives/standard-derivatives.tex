\subsection*{Standard Derivative Table}

\begin{center}
\begin{tabular}{ll}
\toprule
Function & Derivative \\
\midrule
\(x^n\) & \(nx^{n-1}\) \\
\(\sqrt{x}\) & \(1/(2\sqrt{x})\) \\
\(1/x\) & \(-1/x^2\) \\
\(e^x\) & \(e^x\) \\
\(a^x\) & \(a^x\ln a\) \\
\(\ln x\) & \(1/x\) \\
\(\log_a x\) & \(1/(x\ln a)\) \\
\(\sin x\) & \(\cos x\) \\
\(\cos x\) & \(-\sin x\) \\
\(\tan x\) & \(\sec^2 x\) \\
\(\arcsin x\) & \(1/\sqrt{1-x^2}\) \\
\(\arctan x\) & \(1/(1+x^2)\) \\
\bottomrule
\end{tabular}
\end{center}
